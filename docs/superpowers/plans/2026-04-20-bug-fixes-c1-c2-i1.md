void SynthEngine::handleMidiMessage(const juce::MidiMessage& msg)
{
    if (msg.isNoteOn())
    {
        Voice* v = findVoiceForNote(msg.getNoteNumber());
        bool stolen = false;
        if (!v)
        {
            v = findFreeVoice(&stolen);
        }

}

```cpp
Voice* SynthEngine::findFreeVoice(bool* stolen) noexcept
{
    for (auto& v : voices)
        if (!v.isActive()) { if (stolen) *stolen = false; return &v; }

    // All voices busy — steal the one that has been playing longest.
    Voice* oldest = &voices[0];
    for (auto& v : voices)
        if (v.getNoteOnOrder() < oldest->getNoteOnOrder()) oldest = &v;

    if (stolen) *stolen = true;
    return oldest;
}
```

## Task 4 — I-1: osc2 detune LFO not applied

**Bug:** In `Voice::getNextSample()`, when the detune LFO is active, only `osc1.setDetuneCents()` is called. `osc2` keeps its base detune value and doesn't modulate.

**Files:**
- Modify: `Source/Engine/Voice.cpp`
- Test: `Tests/VoiceTests.cpp`

- [ ] **Step 1: Write the failing test**

Replace `Tests/VoiceTests.cpp` with:

```cpp
#include <juce_core/juce_core.h>
#include "Engine/Voice.h"

struct VoiceTests : public juce::UnitTest
{
    VoiceTests() : juce::UnitTest("Voice", "AISynth") {}

    void runTest() override
    {
        beginTest("detune LFO modulates osc2 as well as osc1");

        // Strategy: osc1 level = 0, osc2 level = 1.
        // With the bug, osc2 detune never changes → output is periodic at fixed frequency.
        // With the fix, osc2 detune wobbles → the output changes over LFO cycles.
        // We compare two 512-sample windows 256 samples apart.
        // If osc2 is modulated, they will differ (LFO moved the frequency).
        // If osc2 is NOT modulated, both windows are identical (bug present).

        Voice voice;
        juce::dsp::ProcessSpec spec { 44100.0, 512, 1 };
        voice.prepare(spec);

        VoiceParams p;
        p.osc1.level   = 0.f;   // silence osc1
        p.osc2.level   = 1.f;   // osc2 only
        p.osc2.detune  = 0.f;
        p.ampEnv.attack  = 0.001f;
        p.ampEnv.decay   = 0.001f;
        p.ampEnv.sustain = 1.0f;
        p.ampEnv.release = 0.001f;
        p.fltEnv.attack  = 0.001f;
        p.fltEnv.decay   = 0.001f;
        p.fltEnv.sustain = 1.0f;
        p.fltEnv.release = 0.001f;
        // Fast LFO targeting Detune — cycles in ~22 samples at 2000 Hz
        p.lfo1.shape  = Waveform::Sine;
        p.lfo1.rate   = 2000.f;
        p.lfo1.depth  = 1.0f;
        p.lfo1.dest   = LfoDest::Detune;
        voice.setParams(p);
        voice.noteOn(60, 1.0f);

        // Collect 512 samples; then compare window at offset 0 vs offset 256.
        std::vector<float> buf(512);
        for (auto& s : buf)
            s = voice.getNextSample().first;

        // If osc2 detune is modulated, the 256-sample-later window should differ
        // because the LFO has advanced and is at a different phase.
        float diff = 0.f;
        for (int i = 0; i < 256; ++i)
            diff += std::abs(buf[i] - buf[i + 256]);

        // With LFO at 2000 Hz, even 1 sample shift creates audible difference.
        // Threshold chosen conservatively — any real modulation exceeds 0.01.
        expect(diff > 0.01f,
               "osc2 output must change over time when detune LFO is active (diff=" +
               juce::String(diff) + ")");
    }
};

static VoiceTests voiceTests;
```
